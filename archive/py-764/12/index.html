Тема 12. Створюємо графічний редактор Paint.

Більше про метод .grid() читайте ТУТ

  1. from tkinter import *
  2.  
  3. canvas_width=800
  4. canvas_height=600
  5.  
  6. color = 'black'
  7. stick_size = 3
  8.  
  9. def plus():
  10.     global stick_size
  11.     stick_size+=1
  12. def minus():
  13.     pass
  14.  
  15. def paint(event):
  16.     global color
  17.     global stick_size
  18.  
  19.     x1=event.x - stick_size
  20.     x2=event.x + stick_size
  21.     y1=event.y - stick_size
  22.     y2=event.y + stick_size
  23.     canvas.create_oval(x1,y1,x2,y2, fill=color, outline=color)
  24.  
  25. def change_color(new_color):
  26.     global color
  27.     color=new_color
  28.  
  29. def clear():
  30.     canvas.delete('all')
  31.  
  32. tk = Tk()
  33. tk.resizable(0,0)
  34.  
  35. canvas = Canvas(tk, width = canvas_width, height = canvas_height, bg='white')
  36. canvas.grid(row=2, column=0, columnspan=80)
  37.  
  38. btn_plus = Button(text='+', font=('Comic Sans MS', 15, 'bold'), command=plus, bg='#00e68e', fg='#012340')
  39. btn_plus.grid(row=0, column=3, rowspan=2, columnspan=4, pady=10)
  40. btn_minus = Button(text='-', font=('Comic Sans MS', 15, 'bold'), command=minus, bg='#00e68e', fg='#012340')
  41. btn_minus.grid(row=0, column=7, rowspan=2, columnspan=4, pady=10 )
  42.  
  43. btn_grey = Button(text='', bg='grey', width=2, bd=0, command=lambda:change_color('grey'))
  44. btn_grey.grid(row=0, column=30, sticky=W+E, pady=2, padx=2)
  45. btn_green = Button(text='', bg='green', width=2, bd=0, command=lambda:change_color('green'))
  46. btn_green.grid(row=1, column=30, sticky=W+E, pady=2, padx=2)
  47. canvas.bind('<ButtonPress-1>', paint)
  48.  
  49. canvas.bind('<B1-Motion>', paint)
  50.  
  51. tk.mainloop()
